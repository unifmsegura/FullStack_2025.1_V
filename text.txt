const mongoose = require('mongoose');
const postSchema = new mongoose.Schema({
  titulo: String,
  resumo: String,
  conteudo: String,
});
module.exports = mongoose.model('Post', postSchema);



const express = require('express');
const mongoose = require('mongoose');
const Post = require('./models/Post');
const app = express();

mongoose.connect('mongodb://localhost:27017/meublog');

app.set('view engine', 'ejs');
app.use(express.static('public'));
app.use(express.urlencoded({ extended: true }));

app.get('/', (req, res) => {
  res.sendFile(__dirname + '/projects.html');
});

app.get('/blog', async (req, res) => {
  const posts = await Post.find();
  res.render('blog', { posts });
});

app.get('/cadastrar_post', (req, res) => {
  res.render('cadastrar_post');
});

app.get('/add_post', async (req, res) => {
  const { titulo, resumo, conteudo } = req.query;
  if (titulo && resumo && conteudo) {
    await Post.create({ titulo, resumo, conteudo });
  }
  res.redirect('/blog');
});

app.listen(80, () => console.log('Online na porta 80'));
